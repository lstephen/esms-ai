3   import com.google.common.collect.Sets;
4   import com.ljs.ifootballmanager.ai.Tactic;
5   import com.ljs.ifootballmanager.ai.formation.Formation;
6   import com.ljs.ifootballmanager.ai.player.Player;

8   import java.io.PrintWriter;
9   import java.util.Set;

11  /**
12   *
13   * @author lstephen
14   */
15  public class AtPotentialScorer implements FormationScorer {
16  
17      private Potential potential;
18  
19      private AtPotentialScorer(Potential potential) {
20          this.potential = potential;
21      }
22  
23      public Double score(Formation f, Tactic t) {
24          return DefaultScorer.get().score(atPotential(f), t);
25      }
26  
27      public Double scoring(Formation f, Tactic t) {
28          return DefaultScorer.get().scoring(atPotential(f), t);
29      }
30  
31      public Double defending(Formation f, Tactic t) {
32          return DefaultScorer.get().defending(atPotential(f), t);
33      }
34  
35      public void print(Formation f, PrintWriter w) { }
36  
37      private Formation atPotential(Formation f) {
38          Set<Player> updated = Sets.newHashSet();
39          for (Player p : f.unsortedPlayers()) {
40              updated.add(potential.atPotential(p));
41          }
42          return f.withUpdatedPlayers(updated);
43      }
44  
45      public static AtPotentialScorer create(Potential potential) {
46          return new AtPotentialScorer(potential);
47      }
48  
49  }
