1   package com.ljs.ifootballmanager.ai.value.impl;
2   
3   import com.ljs.ifootballmanager.ai.player.Player;
4   import com.ljs.ifootballmanager.ai.value.Value;
5   
6   /**
7    *
8    * @author lstephen
9    */
10  public class EflValue implements Value {
11  
12      public Double getValue(Player p) {
13          return p.getOverall().getRating()
14              + getYearsLeft(p)
15              + Math.max(22 - p.getAge(), 0);
16      }
17  
18      private Integer getYearsLeft(Player p) {
19          return getRetirementAge(p) - p.getAge();
20      }
21  
22      private Integer getRetirementAge(Player p) {
23          switch (p.getPrimarySkill()) {
24              case STOPPING: return 38;
25              case TACKLING: return 37;
26              case PASSING:
27              case SHOOTING:
28                  return 36;
29              default:
30                  throw new IllegalArgumentException(p.getPrimarySkill().toString());
31          }
32      }
33  
34      public static EflValue create() {
35          return new EflValue();
36      }
37  }
