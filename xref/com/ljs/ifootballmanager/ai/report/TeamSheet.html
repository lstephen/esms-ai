1   package com.ljs.ifootballmanager.ai.report;
2   
3   import com.google.common.base.Function;
4   import com.google.common.collect.ImmutableList;
5   import com.google.common.collect.Iterables;
6   import com.ljs.ifootballmanager.ai.formation.Formation;
7   import com.ljs.ifootballmanager.ai.player.Player;
8   import com.ljs.ifootballmanager.ai.selection.Bench;
9   import com.ljs.ifootballmanager.ai.selection.ChangePlan;
10  import java.io.PrintWriter;
11  
12  /**
13   *
14   * @author lstephen
15   */
16  public class TeamSheet implements Report {
17  
18      private final String team;
19  
20      private final Formation formation;
21  
22      private final ChangePlan changePlan;
23  
24      private final Bench bench;
25  
26      private TeamSheet(String team, Formation formation, ChangePlan cp, Bench bench) {
27          this.team = team;
28          this.formation = formation;
29          this.changePlan = cp;
30          this.bench = bench;
31      }
32  
33      private ImmutableList<Player> players() {
34          return ImmutableList.copyOf(Iterables.concat(formation.players(), bench.players()));
35      }
36  
37      private Function<Player, Integer> getPlayerIndex() {
38          return new Function<Player, Integer>() {
39              public Integer apply(Player p) {
40                  return players().indexOf(p) + 1;
41              }
42          };
43      }
44  
45      public void print(PrintWriter w) {
46          w.println(team);
47          w.println(formation.getTactic().getCode());
48          w.println();
49  
50          formation.printPlayers(w);
51          w.println();
52          bench.printPlayers(w);
53          w.println();
54          w.format("PK: %s%n", formation.getPenaltyKicker().getName());
55          w.println();
56          changePlan.print(w, getPlayerIndex());
57      }
58  
59  
60      public static TeamSheet create(String team, Formation formation, ChangePlan cp, Bench bench) {
61          return new TeamSheet(team, formation, cp, bench);
62      }
63  
64  }
